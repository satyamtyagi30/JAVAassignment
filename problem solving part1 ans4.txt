Q4. Create subclass Cylinder which is derived from the Circle

♦ Cylinder class contains one instance variable i.e height (Type = double)

♦ Constructors in the cylinder class and also invoke the constructor of Circle class

♦ Invoke the circle class variables and methods by the instance of the Cylinder class

♦ Create one public method: getVolume().

write a test program to test Cylinder class and its features.
ANS:


//file name: Circle.java
package Classtest;

public class Circle
{  
 
final double PI=3.14;

//2 PRIVATE DATA MEMBERS

 private double radius=1.0; 
 private String color="red";  
 
//2 OVERLOADING CONSTRUCTOR

	Circle()
	    {
              this.radius=1.0;
	    }
	
	Circle(double radius,String color)
	    {
              this.radius = radius;
              this.color  = color; 
	    }

//2 PUBLIC FUNCTIONS

	public void getradius()
                   {
			System.out.println("Radius: "+radius);
                   }
        public void getArea()
		{
			double area= PI * radius * radius;
			System.out.println("Area : "+area);             
 		}

 }  


public class Cylinder extends Circle
{
final double Pi=3.14;
  double height,volume;
   Cylinder()
	{
	  height= 10;
	}


   Cylinder(int h,double radius,String color)
	{
	  
	  super(radius,color);
	  height= h;
	}

	// invoking the instance variable of parent class
    height = super.radius;

   void getradius()
    {
        // calling the overriden method
        super.getradius();
        System.out.println("This is the Cylinder class");
        
        // printing the name
        System.out.println(height);
    }

	public void getVolume(double ht)
		{
		   volume= Pi * super.radius * super.radius * ht ;
		   System.out.println("Volume: "+volume);
		}
  
}

class  A
{
   public static void main(String[] args)
{
   // instance of the derived class
        Cylinder cy = new Cylinder();
	
        
        // calling the unoverriden method print
        cy.getradius();
	cy.getVolume(20);

}

}
